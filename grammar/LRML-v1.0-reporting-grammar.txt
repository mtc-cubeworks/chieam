LRML = "lrml", "model", Identifier, "{",           { ImportDef },           { DatasetDef | MetricDef | ReportDef | DashboardDef | StyleDef },       "}" ;  ImportDef = "import", String, [ "as", Identifier ], ";" ;  --------------------------------------------------------- -- DATASETS (Semantic Connections) ---------------------------------------------------------  DatasetDef = "dataset", Identifier, "{",                 { SourceRef },                 [ FilterBlock ],                 [ GroupBlock ],                 [ ComputeBlock ],              "}" ;  SourceRef = "source", "=", DSLSource, ";" ;  DSLSource =       "dbml",  ":", QualifiedName     | "uxml",  ":", QualifiedName     | "wdml",  ":", QualifiedName     | "sdml",  ":", QualifiedName     | "aiml",  ":", QualifiedName     | "adml",  ":", QualifiedName     | "blml",  ":", QualifiedName     | "sql",   ":", String        (* Custom SQL if needed *)     ;  FilterBlock = "filter", "{", { FilterExpr ";" }, "}" ; FilterExpr  = Identifier, FilterOp, Scalar ; FilterOp    = "==" | "!=" | "<" | ">" | "<=" | ">=" | "contains" ;  GroupBlock = "groupBy", "{", { Identifier ";" }, "}" ;  ComputeBlock = "compute", "{", { ComputeExpr ";" }, "}" ; ComputeExpr  = Identifier, "=", ComputeFunc ;  ComputeFunc =       "count" "(" Identifier ")"     | "sum"   "(" Identifier ")"     | "avg"   "(" Identifier ")"     | "min"   "(" Identifier ")"     | "max"   "(" Identifier ")"     | Identifier ;  --------------------------------------------------------- -- METRICS ---------------------------------------------------------  MetricDef = "metric", Identifier, "{",                 "dataset", "=", Identifier, ";",                 "value",   "=", MetricExpr, ";",                 [ "label", "=", String, ";" ],                 [ "format", "=", String, ";" ],             "}" ;  MetricExpr = ComputeFunc | Identifier ;  --------------------------------------------------------- -- REPORT ---------------------------------------------------------  ReportDef = "report", Identifier, "{",                 "kind", "=", ReportKind, ";",                  (* Data sources *)                 { "dataset", "=", Identifier, ";" },                 { "metric",  "=", Identifier, ";" },                  (* Print Profile Optional *)                 [ PrintProfile ],                  (* Sections *)                 { SectionDef },                  "output", "=", OutputKind, ";",             "}" ;  ReportKind =       "tabular"     | "summary"     | "audit"     | "compliance"     | "ml-benchmark"     | "security"     | "ux"     | "domain"     | "algorithm"     | "ai-agent"     | "print-layout"     | Identifier ;  OutputKind = "pdf" | "html" | "json" | "md" | Identifier ;  --------------------------------------------------------- -- SECTION ---------------------------------------------------------  SectionDef = "section", Identifier, "{",                 [ "title", "=", String, ";" ],                 [ "page-break-before", "=", Boolean, ";" ],                 [ "page-break-after", "=", Boolean, ";" ],                 { SectionContent },              "}" ;  SectionContent =       TableBlock     | NarrativeBlock     | ChartBlock     | ListBlock     | LayoutBlock     ;  --------------------------------------------------------- -- TABLES ---------------------------------------------------------  TableBlock = "table", "{",                 [ "header-repeat", "=", Boolean, ";" ],                 { ColumnDef },              "}" ;  ColumnDef = "column", Identifier, "{",                [ "label", "=", String, ";" ],                [ "width", "=", Measure, ";" ],                [ "align", "=", AlignType, ";" ],             "}"           | "column", Identifier, ";" ;  AlignType = "left" | "center" | "right" ;  --------------------------------------------------------- -- NARRATIVE ---------------------------------------------------------  NarrativeBlock = "narrative", "{",                     "text", "=", String, ";",                  "}" ;  --------------------------------------------------------- -- CHARTS ---------------------------------------------------------  ChartBlock = "chart", "{",                 "kind", "=", ChartKind, ";",                 "dimension", "=", Identifier, ";",                 "measure",   "=", Identifier, ";",              "}" ;  ChartKind = "bar" | "line" | "pie" | "area" | "kpi" | Identifier ;  --------------------------------------------------------- -- LIST ---------------------------------------------------------  ListBlock = "list", "{", { "item", "=", String, ";" }, "}" ;  --------------------------------------------------------- -- DASHBOARD ---------------------------------------------------------  DashboardDef = "dashboard", Identifier, "{",                   { DashTile },                "}" ;  DashTile = "tile", Identifier, "{",                 "metric", "=", Identifier, ";",                 [ "size",  "=", String, ";" ],                 [ "label", "=", String, ";" ],            "}" ;  --------------------------------------------------------- -- PRINT LAYOUT (Absolute positioning + page setup) ---------------------------------------------------------  PrintProfile = "print", "{",                   [ "page-size",   "=", PageSize, ";" ],                   [ "orientation", "=", Orientation, ";" ],                   [ "margin",      "=", MarginSpec, ";" ],                   { PrintExtra },                "}" ;  PageSize     = "A4" | "Letter" | "Legal" | "A5" | String ; Orientation  = "portrait" | "landscape" ; MarginSpec   = Measure | String ;  PrintExtra =       LayoutBlock     | StyleDef ;  --------------------------------------------------------- -- LAYOUT BLOCKS (ABSOLUTE / GRID) ---------------------------------------------------------  LayoutBlock =       BoxBlock     | GridBlock     | PositionBlock ;  (* ABSOLUTE BOX *) BoxBlock = "box", Identifier, "{",               [ "x",      "=", Measure, ";" ],               [ "y",      "=", Measure, ";" ],               [ "width",  "=", Measure, ";" ],               [ "height", "=", Measure, ";" ],               [ "style",  "=", QualifiedName, ";" ],               { LayoutContent },           "}" ;  (* GRID LAYOUT *) GridBlock = "grid", Identifier, "{",                "columns", "=", "[", { Measure }, "]", ";",                "rows",    "=", "[", { Measure }, "]", ";",                { GridCell },             "}" ;  GridCell = "cell", Identifier, "{",                "col", "=", Number, ";",                "row", "=", Number, ";",                { LayoutContent },           "}" ;  (* POSITIONAL BLOCK *) PositionBlock = "position", Identifier, "{",                    "x", "=", Measure, ";",                    "y", "=", Measure, ";",                    { LayoutContent },                 "}" ;  LayoutContent =       "text", "=", String, ";"     | "bind", "=", Identifier, ";"     | TableBlock     | NarrativeBlock ;  --------------------------------------------------------- -- STYLE DEFINITIONS (TOKENS / WDML Hooks) ---------------------------------------------------------  StyleDef = "style", Identifier, "{",                [ "font",       "=", QualifiedName, ";" ],                [ "font-size",  "=", Measure, ";" ],                [ "bold",       "=", Boolean, ";" ],                [ "italic",     "=", Boolean, ";" ],                [ "color",      "=", QualifiedName, ";" ],                [ "background", "=", QualifiedName, ";" ],                [ "border",     "=", BorderSpec, ";" ],                [ "padding",    "=", Measure, ";" ],             "}" ;  BorderSpec = Measure | String ;  --------------------------------------------------------- -- MEASURES & LEXICAL ---------------------------------------------------------  Measure = Number "mm"         | Number "px"         | Number "in"         | Number "%"         ;  Identifier    = letter, { letter | digit | "_" } ; QualifiedName = Identifier, { ".", Identifier } ; String        = '"', { character }, '"' ; Number        = digit, { digit }, [ ".", { digit } ] ; Boolean       = "true" | "false" ;
